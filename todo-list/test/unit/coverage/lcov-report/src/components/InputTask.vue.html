import { Task } from '../models/Task'
 
export default {
  name: 'InputTask',
  data () {
    return {
    }
  },
  methods: {
    addTask ($event) {
      let value = $event.target.value
      let task = this.createTask(value)
      this.broadcast(task)
      this.clearField($event)
    },
    createTask (value) {
      let task = new Task()
      task.completed = false
      task.title = value
      return task
    },
    clearField () {
      this.$el.querySelector('input').value = ''
    },
    broadcast (task) {
      this.$emit('newTask', task)
    }
  }
}
